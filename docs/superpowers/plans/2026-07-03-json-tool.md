# JSON 工具实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在工具箱中新增 JSON 工具，提供格式化、压缩、转义、反转义、校验等功能

**Architecture:** 继承 BaseFrame 创建 JsonToolFrame 窗口类，复用现有窗口框架模式；渲染进程创建 Vue 组件实现左右分屏布局；通过 IPC 通信实现主进程功能调用

**Tech Stack:** Electron + Vue 3 + TypeScript + BaseFrame 窗口框架

## Global Constraints

- 窗口必须继承 BaseFrame 基类
- IPC 频道命名遵循 `to-main-{窗口名}:{方法名}` 格式
- 样式使用 Modern Dark (Cinema) 风格，Slate/Blue 配色方案
- 跨平台兼容：Windows 和 macOS
- 代码修改后必须执行 `npm run typecheck` 检查

```typescript
import { BrowserWindowConstructorOptions, ipcMain, dialog } from 'electron'
import BaseFrame from './BaseFrame'

/**
 * JSON 工具窗口
 * @description 提供 JSON 格式化、压缩、转义、反转义、校验等功能
 */
export default class JsonToolFrame extends BaseFrame {
  /** 窗口宽度 */
  static readonly WIDTH = 900

  /** 窗口高度 */
  static readonly HEIGHT = 600

  /** 最小宽度 */
  static readonly MIN_WIDTH = 600

  /** 最小高度 */
  static readonly MIN_HEIGHT = 400

  /** 窗口配置 - 透明无边框 */
  protected readonly options: BrowserWindowConstructorOptions = {
    width: JsonToolFrame.WIDTH,
    height: JsonToolFrame.HEIGHT,
    minWidth: JsonToolFrame.MIN_WIDTH,
    minHeight: JsonToolFrame.MIN_HEIGHT,
    transparent: true,
    backgroundColor: '#00000000',
    frame: false,
    show: false,
    resizable: true
  }

  /** 路由路径 */
  protected readonly routePath: string = '/jsonTool'

  /** 是否已注册 IPC */
  static #ipcRegistered = false

  /**
   * 注册 IPC 监听器
   */
  protected registerIPC(): void {
    super.registerIPC()

    // 只注册一次 IPC
    if (JsonToolFrame.#ipcRegistered) {
      return
    }

    // 最小化窗口
    ipcMain.on('to-main-JsonTool:minimize', () => {
      if (this.isAlive()) {
        this.window!.minimize()
      }
    })

    // 关闭窗口
    ipcMain.on('to-main-JsonTool:close', () => {
      this.close()
    })

    // 格式化 JSON
    ipcMain.handle('to-main-JsonTool:format', (_event, input: string) => {
      try {
        const parsed = JSON.parse(input)
        return { success: true, result: JSON.stringify(parsed, null, 2) }
      } catch (e) {
        return { success: false, error: (e as Error).message }
      }
    })

    // 压缩 JSON
    ipcMain.handle('to-main-JsonTool:compress', (_event, input: string) => {
      try {
        const parsed = JSON.parse(input)
        return { success: true, result: JSON.stringify(parsed) }
      } catch (e) {
        return { success: false, error: (e as Error).message }
      }
    })

    // 字符串转义
    ipcMain.handle('to-main-JsonTool:escape', (_event, input: string) => {
      try {
        const escaped = input
          .replace(/\\/g, '\\\\')
          .replace(/"/g, '\\"')
          .replace(/\n/g, '\\n')
          .replace(/\r/g, '\\r')
          .replace(/\t/g, '\\t')
        return { success: true, result: escaped }
      } catch (e) {
        return { success: false, error: (e as Error).message }
      }
    })

    // 字符串反转义
    ipcMain.handle('to-main-JsonTool:unescape', (_event, input: string) => {
      try {
        const unescaped = input
          .replace(/\\n/g, '\n')
          .replace(/\\r/g, '\r')
          .replace(/\\t/g, '\t')
          .replace(/\\"/g, '"')
          .replace(/\\\\/g, '\\')
        return { success: true, result: unescaped }
      } catch (e) {
        return { success: false, error: (e as Error).message }
      }
    })

    // 校验 JSON
    ipcMain.handle('to-main-JsonTool:validate', (_event, input: string) => {
      try {
        JSON.parse(input)
        return { success: true, valid: true }
      } catch (e) {
        return { success: true, valid: false, error: (e as Error).message }
      }
    })

    JsonToolFrame.#ipcRegistered = true
  }
}
```

/**
 * 获取 JSON 工具窗口
 * @returns JsonToolFrame 实例
 */
getJsonToolFrame(): JsonToolFrame {
  if (!this.#jsonToolFrame) {
    this.#jsonToolFrame = new JsonToolFrame()
    this.#jsonToolFrame.onDestroyCallback = () => {
      this.#jsonToolFrame = null
    }
  }
  return this.#jsonToolFrame
}

/**
 * 创建 JSON 工具窗口
 * @returns JsonToolFrame 实例
 */
createJsonToolFrame(): JsonToolFrame {
  const frame = this.getJsonToolFrame()
  frame.create()
  return frame
}

<script setup lang="ts">
import { ref, onMounted, onUnmounted } from 'vue'

const inputText = ref('')
const outputText = ref('')
const hasError = ref(false)
const errorMessage = ref('')
const isValid = ref(false)

/**
 * 格式化 JSON
 */
const format = async (): Promise<void> => {
  if (!inputText.value.trim()) return
  const result = await window.electron.ipcRenderer.invoke('to-main-JsonTool:format', inputText.value)
  handleResult(result)
}

/**
 * 压缩 JSON
 */
const compress = async (): Promise<void> => {
  if (!inputText.value.trim()) return
  const result = await window.electron.ipcRenderer.invoke('to-main-JsonTool:compress', inputText.value)
  handleResult(result)
}

/**
 * 字符串转义
 */
const escape = async (): Promise<void> => {
  if (!inputText.value.trim()) return
  const result = await window.electron.ipcRenderer.invoke('to-main-JsonTool:escape', inputText.value)
  handleResult(result)
}

/**
 * 字符串反转义
 */
const unescape = async (): Promise<void> => {
  if (!inputText.value.trim()) return
  const result = await window.electron.ipcRenderer.invoke('to-main-JsonTool:unescape', inputText.value)
  handleResult(result)
}

/**
 * 校验 JSON
 */
const validate = async (): Promise<void> => {
  if (!inputText.value.trim()) return
  const result = await window.electron.ipcRenderer.invoke('to-main-JsonTool:validate', inputText.value)
  if (result.valid) {
    hasError.value = false
    errorMessage.value = ''
    isValid.value = true
  } else {
    hasError.value = true
    errorMessage.value = result.error
    isValid.value = false
  }
}

/**
 * 处理结果
 */
const handleResult = (result: { success: boolean; result?: string; error?: string }): void => {
  if (result.success) {
    outputText.value = result.result || ''
    hasError.value = false
    errorMessage.value = ''
    isValid.value = false
  } else {
    outputText.value = ''
    hasError.value = true
    errorMessage.value = result.error || '操作失败'
    isValid.value = false
  }
}

/**
 * 清空所有内容
 */
const clearAll = (): void => {
  inputText.value = ''
  outputText.value = ''
  hasError.value = false
  errorMessage.value = ''
  isValid.value = false
}

/**
 * 复制结果到剪贴板
 */
const copyResult = async (): Promise<void> => {
  if (outputText.value) {
    await navigator.clipboard.writeText(outputText.value)
  }
}

/**
 * 打开文件
 */
const openFile = async (): Promise<void> => {
  const result = await window.electron.ipcRenderer.invoke('to-main-JsonTool:openFile')
  if (result) {
    inputText.value = result
  }
}

/**
 * 保存结果到文件
 */
const saveResult = async (): Promise<void> => {
  if (outputText.value) {
    await window.electron.ipcRenderer.invoke('to-main-JsonTool:saveFile', outputText.value)
  }
}

/**
 * 处理文件拖拽
 */
const handleDrop = async (event: DragEvent): Promise<void> => {
  const files = event.dataTransfer?.files
  if (files && files.length > 0) {
    const file = files[0]
    if (file.name.endsWith('.json')) {
      const text = await file.text()
      inputText.value = text
    }
  }
}

/**
 * 最小化窗口
 */
const minimize = (): void => {
  window.electron.ipcRenderer.send('to-main-JsonTool:minimize')
}

/**
 * 关闭窗口
 */
const close = (): void => {
  window.electron.ipcRenderer.send('to-main-JsonTool:close')
}

/**
 * 键盘快捷键处理
 */
const handleKeydown = (e: KeyboardEvent): void => {
  if (e.ctrlKey && e.shiftKey) {
    switch (e.key.toUpperCase()) {
      case 'F':
        e.preventDefault()
        format()
        break
      case 'M':
        e.preventDefault()
        compress()
        break
      case 'E':
        e.preventDefault()
        escape()
        break
      case 'U':
        e.preventDefault()
        unescape()
        break
      case 'V':
        e.preventDefault()
        validate()
        break
      case 'DELETE':
        e.preventDefault()
        clearAll()
        break
    }
  }
}

onMounted(() => {
  document.addEventListener('keydown', handleKeydown)
})

onUnmounted(() => {
  document.removeEventListener('keydown', handleKeydown)
})
</script>

      { name: 'All Files', extensions: ['*'] }
    ]
  })
  if (result.canceled || !result.filePath) {
    return false
  }
  const fs = require('fs')
  fs.writeFileSync(result.filePath, content, 'utf-8')
  return true
